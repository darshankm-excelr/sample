# Create an employee class with empname as public, project attribute as protected and salary as a private attribute and create work method inside the class to access these 3 attributes?
 
class Employee:
    def __init__(self, empname, project, salary):
        self.empname = empname
        self._project = project
        self.__salary = salary

    def work(self):
        print(f"Employee Name: {self.empname}")
        print(f"Project: {self._project}")
        print(f"Salary: {self.__salary}")

employee1 = Employee(empname="John", project="ABC Project", salary=50000)

employee1.work()
Employee Name: John
Project: ABC Project
Salary: 50000

# 3 Create an employee class, use Constructor for public variable name, Private variable salary and create a show method to access that variables? 

class Employee:
    def __init__(self, name, salary):
        # Public variable
        self.name = name
        # Private variable
        self.__salary = salary

    def show(self):
        # Accessing both public and private variables
        print(f"Employee Name: {self.name}")
        print(f"Salary: {self.__salary}")

employee1 = Employee(name="John Doe", salary=50000)

employee1.show()
Employee Name: John Doe
Salary: 50000

# 4 Create parent and child class and create a constructor in company with protected variable and use that constructor in child class and also define any method in child class to access that protected variable?
class Company:
    def __init__(self, company_name):
        self._company_name = company_name

class Employee(Company):
    def __init__(self, company_name, employee_name):
        super().__init__(company_name)
        self.employee_name = employee_name

    def show_company_name(self):
        print(f"Employee Name: {self.employee_name}")
        print(f"Company Name: {self._company_name}")

employee = Employee(company_name="ABC Corporation", employee_name="John Doe")
employee.show_company_name()

Employee Name: John Doe
Company Name: ABC Corporation

# 5 Create a parent class as a base with constructor containing private and public variable. Then create a child class by calling Constructor of base class? 
class Parent:
    def __init__(self, private_var, public_var):
        self.__private_var = private_var
        self.public_var = public_var

class Child(Parent):
    def __init__(self, private_var, public_var, child_var):
        super().__init__(private_var, public_var)
        self.child_var = child_var

child_instance = Child(private_var="private_value", public_var="public_value", child_var="child_value")
print("Private Variable:", child_instance._Parent__private_var)
print("Public Variable:", child_instance.public_var)
print("Child Variable:", child_instance.child_var)
Private Variable: private_value
Public Variable: public_value
Child Variable: child_value

# 6 Create a class with private public and protected attributes and private public and protected methods in a single class? 

class MyClass:
    def __init__(self):
        self.__private_attr = "Private Attribute"
        self.public_attr = "Public Attribute"
        self._protected_attr = "Protected Attribute"

    def __private_method(self):
        print("Private Method")

    def public_method(self):
        print("Public Method")

    def _protected_method(self):
        print("Protected Method")

my_instance = MyClass()

print("Private Attribute:", my_instance._MyClass__private_attr)
print("Public Attribute:", my_instance.public_attr)
print("Protected Attribute:", my_instance._protected_attr)

my_instance._MyClass__private_method()
my_instance.public_method()
my_instance._protected_method()
Private Attribute: Private Attribute
Public Attribute: Public Attribute
Protected Attribute: Protected Attribute
Private Method
Public Method
Protected Method

# 8 Create class details with name, age and job as protected attribute. And create another class with function to access all these variables? 
class Details:
    def __init__(self, name, age, job):
        self._name = name
        self._age = age
        self._job = job

class AccessDetails:
    def access_attributes(self, details_object):
        print("Name:", details_object._name)
        print("Age:", details_object._age)
        print("Job:", details_object._job)

details_instance = Details(name="John Doe", age=30, job="Engineer")

access_details_instance = AccessDetails()

access_details_instance.access_attributes(details_instance)
Name: John Doe
Age: 30
Job: Engineer

# 9 Create class with name, age and job as protected attributes. And create another class using Constructor to access these 3 variables?
class Person:
    def __init__(self, name, age, job):
        self._name = name
        self._age = age
        self._job = job

class Employee:
    def __init__(self, person_object):
        self.name = person_object._name
        self.age = person_object._age
        self.job = person_object._job

person_instance = Person(name="John Doe", age=30, job="Engineer")

employee_instance = Employee(person_object=person_instance)

print("Name:", employee_instance.name)
print("Age:", employee_instance.age)
print("Job:", employee_instance.job)
Name: John Doe
Age: 30
Job: Engineer
